## 🌊 PART 8: ENHANCED ONBOARDING (Gentler Streak Inspired)

### **Onboarding Screen with Beautiful Transitions**

```typescript
const OnboardingScreen = ({ step, totalSteps, question, onNext }) => {
  const progressAnim = useRef(new Animated.Value(0)).current;
  const contentOpacity = useRef(new Animated.Value(0)).current;
  
  useEffect(() => {
    // Animate progress bar
    Animated.spring(progressAnim, {
      toValue: (step / totalSteps) * 100,
      useNativeDriver: false,
      tension: 40,
      friction: 8,
    }).start();
    
    // Fade in content
    Animated.timing(contentOpacity, {
      toValue: 1,
      duration: 600,
      useNativeDriver: true,
    }).start();
  }, [step]);

  return (
    <View style={styles.screen}>
      {/* Background with subtle gradient */}
      <LinearGradient
        colors={['#FFFFFF', '#F5F1E8']}
        style={styles.background}
      >
        {/* Progress indicator */}
        <View style={styles.progressContainer}>
          <View style={styles.progressBar}>
            <Animated.View
              style={[
                styles.progressFill,
                {
                  width: progressAnim.interpolate({
                    inputRange: [0, 100],
                    outputRange: ['0%', '100%'],
                  }),
                },
              ]}
            />
          </View>
          
          <Text style={styles.progressText}>
            {step} of {totalSteps}
          </Text>
        </View>

        {/* Content */}
        <Animated.View
          style={[
            styles.content,
            { opacity: contentOpacity }
          ]}
        >
          {/* Question */}
          <Text style={styles.question}>
            {question.title}
          </Text>
          
          {question.subtitle && (
            <Text style={styles.subtitle}>
              {question.subtitle}
            </Text>
          )}

          {/* Options */}
          <View style={styles.options}>
            {question.options.map((option) => (
              <OptionCard
                key={option.id}
                option={option}
                onSelect={onNext}
              />
            ))}
          </View>
        </Animated.View>
      </LinearGradient>
    </View>
  );
};

const OptionCard = ({ option, onSelect }) => {
  const scaleAnim = useRef(new Animated.Value(1)).current;
  
  const handlePress = () => {
    // Animation feedback
    Animated.sequence([
      Animated.timing(scaleAnim, {
        toValue: 0.96,
        duration: 100,
        useNativeDriver: true,
      }),
      Animated.timing(scaleAnim, {
        toValue: 1,
        duration: 100,
        useNativeDriver: true,
      }),
    ]).start(() => {
      onSelect(option);
    });
    
    // Haptic feedback
    Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Light);
  };

  return (
    <Animated.View style={{ transform: [{ scale: scaleAnim }] }}>
      <TouchableOpacity
        style={styles.optionCard}
        activeOpacity={1}
        onPress={handlePress}
      >
        <BlurView
          blurType="light"
          blurAmount={8}
          style={styles.optionBlur}
        >
          <View style={styles.optionContent}>
            {option.emoji && (
              <Text style={styles.optionEmoji}>{option.emoji}</Text>
            )}
            <Text style={styles.optionText}>{option.text}</Text>
          </View>
        </BlurView>
      </TouchableOpacity>
    </Animated.View>
  );
};

const styles = StyleSheet.create({
  screen: {
    flex: 1,
  },
  
  background: {
    flex: 1,
    paddingHorizontal: 24,
    paddingTop: 60,
  },
  
  progressContainer: {
    marginBottom: 48,
  },
  
  progressBar: {
    height: 4,
    backgroundColor: '#E8DFD0',
    borderRadius: 2,
    marginBottom: 12,
    overflow: 'hidden',
  },
  
  progressFill: {
    height: '100%',
    backgroundColor: '#6B9F9A',
    borderRadius: 2,
  },
  
  progressText: {
    fontSize: 13,
    fontWeight: '500',
    letterSpacing: 0.3,
    color: '#5A6C7D',
    textAlign: 'center',
  },
  
  content: {
    flex: 1,
  },
  
  question: {
    fontSize: 28,
    fontWeight: '700',
    letterSpacing: 0.8,
    lineHeight: 36,
    color: '#2C3E50',
    marginBottom: 12,
  },
  
  subtitle: {
    fontSize: 17,
    letterSpacing: 0.3,
    lineHeight: 26,
    color: '#5A6C7D',
    marginBottom: 32,
  },
  
  options: {
    gap: 12,
  },
  
  optionCard: {
    borderRadius: 16,
    overflow: 'hidden',
    borderWidth: 1.5,
    borderColor: 'rgba(107, 159, 154, 0.2)',
  },
  
  optionBlur: {
    backgroundColor: 'rgba(245, 241, 232, 0.5)',
  },
  
  optionContent: {
    flexDirection: 'row',
```
