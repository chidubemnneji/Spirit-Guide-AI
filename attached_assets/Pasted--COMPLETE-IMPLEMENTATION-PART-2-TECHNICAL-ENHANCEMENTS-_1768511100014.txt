CREATE TABLE practices_log (

  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_practices_user ON practices_log(user_id, created_at DESC);

CREATE TABLE emotional_checkins (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

  emotional_state VARCHAR(50),
  intensity INTEGER, -- 1-10
  context TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_emotions_user ON emotional_checkins(user_id, created_at DESC);

  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

);

CREATE INDEX idx_patterns_user ON pattern_analysis(user_id, created_at DESC);

  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

## 🧠 ENHANCEMENT 1: Memory Extraction Service

**File:** `/server/services/memoryExtractor.js`

```javascript
const anthropic = require('./anthropicService');
const db = require('./databaseService');

class MemoryExtractor {
  /**
   * Extract insights from conversation for future reference
   */
  async extractMemories(conversationId, userId) {
    try {
      // Get conversation messages
      const messages = await db.query(
        `SELECT role, content FROM messages 
         WHERE conversation_id = $1 
         ORDER BY created_at ASC`,
        [conversationId]
      );

      if (messages.rows.length < 4) {
        // Too short to extract meaningful insights
        return null;
      }

      const conversationText = messages.rows
        .map(m => `${m.role}: ${m.content}`)
        .join('\n\n');

      // Ask Claude to extract insights
      const response = await anthropic.messages.create({
        model: 'claude-sonnet-4-20250514',
        system: this.getExtractionPrompt(),
        messages: [{
          role: 'user',
          content: `Extract key insights from this spiritual conversation:\n\n${conversationText}`
        }],
        max_tokens: 1500
      });

      const insights = this.parseInsights(response.content[0].text);

      // Save to database
      await this.saveInsights(insights, userId, conversationId);

      return insights;

    } catch (error) {
      console.error('Memory extraction error:', error);
      return null;
    }
  }

  getExtractionPrompt() {
    return `You are analyzing a spiritual conversation to extract key insights.

Extract and return ONLY valid JSON (no markdown, no explanation):

{
  "topics": ["doubt", "prayer"],
  "emotional_state": "struggling",
  "memorable_moments": [
    {
      "type": "breakthrough",
      "summary": "User realized they're not alone",
      "emotional_impact": "relieved"
    }
  ],
  "practices_mentioned": [
    {
      "practice": "journaling",
      "user_attitude": "willing_to_try",
      "barriers": "lack of time"
    }
  ],
  "recurring_themes": ["feels_alone", "questions_faith"],
  "progress_indicators": "More open about doubt than before"
}

Be concise. Focus on what matters for continuing this person's journey.`;
  }

  parseInsights(text) {
    try {
      const cleaned = text.replace(/```json|```/g, '').trim();
      return JSON.parse(cleaned);
    } catch (error) {
      console.error('Failed to parse insights:', error);
      return null;
    }
  }

  async saveInsights(insights, userId, conversationId) {
    if (!insights) return;

    try {
      // Save topics
      if (insights.topics && Array.isArray(insights.topics)) {
        for (const topic of insights.topics) {
          await db.query(
            `INSERT INTO conversation_topics (user_id, topic, sentiment)
             VALUES ($1, $2, $3)
             ON CONFLICT (user_id, topic) 
             DO UPDATE SET 
               last_mentioned = NOW(),
               mention_count = conversation_topics.mention_count + 1,
               sentiment = $3`,
            [userId, topic, insights.emotional_state || 'neutral']
          );
        }
      }

      // Save memorable moments
      if (insights.memorable_moments && Array.isArray(insights.memorable_moments)) {
        for (const moment of insights.memorable_moments) {
          await db.query(
            `INSERT INTO memorable_moments 
             (user_id, conversation_id, moment_type, summary, emotional_state)
             VALUES ($1, $2, $3, $4, $5)`,
            [
              userId,
              conversationId,
              moment.type || 'insight',
              moment.summary,
              moment.emotional_impact || 'neutral'
            ]
          );
        }
      }

      // Save practice mentions
      if (insights.practices_mentioned && Array.isArray(insights.practices_mentioned)) {
        for (const practice of insights.practices_mentioned) {
          await db.query(
            `INSERT INTO practices_log 
             (user_id, practice_type, suggested_by_ai)
             VALUES ($1, $2, $3)`,
            [userId, practice.practice, true]
          );
        }
      }

    } catch (error) {
      console.error('Failed to save insights:', error);
    }
  }
}

module.exports = new MemoryExtractor();
```

```javascript
const anthropic = require('./anthropicService');
const db = require('./databaseService');

class EmotionalIntelligence {
  /**
   * Detect emotional state from message
   */
  async detectEmotion(message, recentHistory = []) {
    try {
      const context = recentHistory.length > 0
        ? `Recent conversation:\n${recentHistory.join('\n')}\n\nCurrent message: ${message}`
        : `Message: ${message}`;

      const response = await anthropic.messages.create({
        model: 'claude-sonnet-4-20250514',
        system: `Detect emotional state. Return ONLY valid JSON (no markdown):

{
  "primary_emotion": "struggling|peaceful|hopeful|overwhelmed|guilty|angry|confused|lonely",
  "intensity": 1-10,
  "urgency": "low|medium|high|crisis",
  "needs": ["validation", "comfort", "guidance", "celebration", "space"],
  "tone_recommendation": "gentle|celebratory|compassionate|thoughtful"
}`,
        messages: [{ role: 'user', content: context }],
        max_tokens: 300
      });

      const cleaned = response.content[0].text.replace(/```json|```/g, '').trim();
      return JSON.parse(cleaned);

    } catch (error) {
      console.error('Emotion detection error:', error);
      return {
        primary_emotion: 'unknown',
        intensity: 5,
        urgency: 'low',
        needs: ['validation'],
        tone_recommendation: 'gentle'
      };
    }
  }

  /**
   * Save emotional check-in
   */
  async saveEmotionalCheckin(userId, emotion, message) {
    try {
      await db.query(
        `INSERT INTO emotional_checkins (user_id, emotional_state, intensity, context)
         VALUES ($1, $2, $3, $4)`,
        [userId, emotion.primary_emotion, emotion.intensity, message.substring(0, 200)]
      );
    } catch (error) {
      console.error('Failed to save emotional checkin:', error);
    }
  }

  /**
   * Build emotional modifier for prompt
   */
  buildEmotionalModifier(emotion) {
    const modifiers = {
      struggling: {
        instruction: 'User is struggling. Be EXTRA gentle. Validate heavily. Don\'t rush to fix. Slower pacing. More space.',
        tone: 'compassionate, patient, present'
      },
      overwhelmed: {
        instruction: 'User is overwhelmed. Keep responses SHORT (2-3 sentences max). Simple language. Grounding, not expanding. No new concepts.',
        tone: 'calm, grounding, minimal words'
      },
      peaceful: {
        instruction: 'User is in a good place. Celebrate this! Light tone okay. Can explore more deeply. Affirm the peace.',
        tone: 'warm, affirming, conversational'
      },
      hopeful: {
        instruction: 'User is feeling hopeful. Nurture this gently. Don\'t over-inflate. Affirm progress. Look forward WITH them.',
        tone: 'encouraging, forward-looking, supportive'
      },
      guilty: {
        instruction: 'User feels guilty. CRITICAL: No "shoulds". Emphasize GRACE heavily. Separate guilt from identity.',
        tone: 'grace-saturated, shame-free, gentle'
      },
      angry: {
        instruction: 'User is angry. Validate the anger. Don\'t minimize. Let them express it. God can handle their anger too.',
        tone: 'validating, steady, non-defensive'
      },
      lonely: {
        instruction: 'User feels alone. Emphasize presence. "You\'re not alone" - but say it genuinely. Point toward connection.',
        tone: 'present, companionable, warm'
      },
      confused: {
        instruction: 'User is confused. Be clear and simple. Help them sort through thoughts. Questions help here.',
        tone: 'clarifying, patient, thoughtful'
      }
    };

    const modifier = modifiers[emotion.primary_emotion] || modifiers.struggling;

    let output = `
═══════════════════════════════════════════════════════════
EMOTIONAL STATE DETECTED
═══════════════════════════════════════════════════════════

Current emotion: ${emotion.primary_emotion} (intensity: ${emotion.intensity}/10)
Urgency: ${emotion.urgency}
User needs: ${emotion.needs.join(', ')}

RESPONSE ADJUSTMENT:
${modifier.instruction}

Recommended tone: ${modifier.tone}
`;

    if (emotion.urgency === 'crisis') {
      output += `
⚠️  CRISIS DETECTED - See crisis protocol
`;
    }

    return output;
  }
}

module.exports = new EmotionalIntelligence();
```
