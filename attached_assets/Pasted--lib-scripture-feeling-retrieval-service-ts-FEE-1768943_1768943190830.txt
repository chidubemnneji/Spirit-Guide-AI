// ============================================
// FEELING-BASED SCRIPTURE RETRIEVAL SERVICE
// RAG-powered retrieval for 6 predefined feelings
// Output format ready for Reflect AI module
// ============================================

import { createClient } from '@supabase/supabase-js';

// ============================================
// TYPES
// ============================================

export type Feeling = 'anxious' | 'sad' | 'grateful' | 'lonely' | 'hopeful' | 'reflective';

export interface ScriptureResult {
  book: string;
  chapter: number;
  verse_start: number;
  verse_end: number | null;
  text: string;
}

export interface UserContext {
  previousLiked?: string[];  // Citations user previously engaged with
  preferredBooks?: string[];
}

interface RankedScripture extends ScriptureResult {
  _similarity: number;
  _topicMatch: number;
  _userBoost: number;
  _totalScore: number;
}

// ============================================
// FEELING → TOPIC MAPPING
// ============================================

const FEELING_TOPIC_MAP: Record<Feeling, {
  topics: string[];
  searchQuery: string;
}> = {
  anxious: {
    topics: ['trust', 'peace', 'fear', 'worry', 'calm', 'rest', 'protection'],
    searchQuery: 'do not be anxious peace trust God fear not cast your cares'
  },
  sad: {
    topics: ['comfort', 'grief', 'sorrow', 'tears', 'mourning', 'healing', 'presence'],
    searchQuery: 'comfort those who mourn brokenhearted weeping sorrow heal'
  },
  grateful: {
    topics: ['thanksgiving', 'praise', 'blessing', 'gratitude', 'joy', 'worship'],
    searchQuery: 'give thanks praise the Lord grateful blessing worship rejoice'
  },
  lonely: {
    topics: ['presence', 'companionship', 'belonging', 'comfort', 'never alone', 'with you'],
    searchQuery: 'I am with you never leave nor forsake not alone presence'
  },
  hopeful: {
    topics: ['hope', 'future', 'promise', 'faith', 'expectation', 'trust', 'plans'],
    searchQuery: 'hope future plans prosper faith confidence promise trust Lord'
  },
  reflective: {
    topics: ['wisdom', 'understanding', 'meditation', 'heart', 'examine', 'truth', 'guidance'],
    searchQuery: 'search my heart wisdom understanding meditate word light path'
  }
};

const VALID_FEELINGS: Feeling[] = ['anxious', 'sad', 'grateful', 'lonely', 'hopeful', 'reflective'];

// ============================================
// SUPABASE CLIENT
// ============================================

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL!,
  process.env.SUPABASE_SERVICE_ROLE_KEY!
);

// ============================================
// EMBEDDING SERVICE
// ============================================

async function getEmbedding(text: string): Promise<number[]> {
  const response = await fetch('https://api.openai.com/v1/embeddings', {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${process.env.OPENAI_API_KEY}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      input: text,
      model: 'text-embedding-3-small',
    }),
  });

  if (!response.ok) {
    throw new Error(`Embedding API error: ${response.statusText}`);
  }

  const data = await response.json();
  return data.data[0].embedding;
}

// ============================================
// MAIN RETRIEVAL FUNCTION
// ============================================

export async function getScripturesByFeeling(
  feeling: Feeling,
  userContext?: UserContext
): Promise<ScriptureResult[]> {
  
  // Validate feeling
  if (!VALID_FEELINGS.includes(feeling)) {
    throw new Error(`Invalid feeling. Must be one of: ${VALID_FEELINGS.join(', ')}`);
  }
  
  const config = FEELING_TOPIC_MAP[feeling];
  
  try {
    // Step 1: Get embedding for semantic search
    const embedding = await getEmbedding(config.searchQuery);
    
    // Step 2: Retrieve candidates via vector similarity
    const { data: vectorResults, error: vectorError } = await supabase.rpc('search_scriptures', {
      query_embedding: embedding,
      match_count: 20, // Get extra for filtering/ranking
      topic_filter: null
    });
    
    if (vectorError) throw vectorError;
    
    // Step 3: Also get topic-matched scriptures
    const { data: topicResults, error: topicError } = await supabase
      .from('scriptures')
      .select('*')
      .overlaps('topics', config.topics)
      .limit(20);
    
    if (topicError) throw topicError;
    
    // Step 4: Merge and deduplicate
    const allResults = mergeResults(vectorResults || [], topicResults || []);
    
    // Step 5: Rank results
    const ranked = rankResults(allResults, config.topics, userContext);
    
    // Step 6: Return top 5, formatted for Reflect module
    return ranked.slice(0, 5).map(formatForOutput);
    
  } catch (error) {
    console.error('Scripture retrieval error:', error);
    
    // Fallback to verified scriptures if RAG fails
    return getFallbackScriptures(feeling);
  }
}

// ============================================
// MERGE RESULTS
// ============================================

function mergeResults(
  vectorResults: any[],
  topicResults: any[]
): any[] {
  const seen = new Set<string>();
  const merged: any[] = [];
  
  // Add vector results first (they have similarity scores)
  for (const result of vectorResults) {
    const key = result.reference || `${result.book}-${result.chapter}-${result.verse_start}`;
    if (!seen.has(key)) {
      seen.add(key);
      merged.push({
        ...result,
        _fromVector: true,
        _similarity: result.similarity || 0
      });
    }
  }
  
  // Add topic results that weren't in vector results
  for (const result of topicResults) {
    const key = result.reference || `${result.book}-${result.chapter}-${result.verse_start}`;
    if (!seen.has(key)) {
      seen.add(key);
      merged.push({
        ...result,
        _fromVector: false,
        _similarity: 0.5 // Default similarity for topic-only matches
      });
    }
  }
  
  return merged;
}

// ============================================
// RANK RESULTS
// ============================================

function rankResults(
  results: any[],
  topics: string[],
  userContext?: UserContext
): RankedScripture[] {
  
  return results.map(result => {
    // Semantic similarity score (0-1)
    const similarity = result._similarity || 0;
    
    // Topic match score (0-1)
    const resultTopics = result.topics || [];
    const matchingTopics = resultTopics.filter((t: string) => 
      topics.includes(t.toLowerCase())
    ).length;
    const topicMatch = topics.length > 0 ? matchingTopics / topics.length : 0;
    
    // User preference boost (0-0.2)
    let userBoost = 0;
    if (userContext) {
      // Boost if user previously liked scriptures from same book
      if (userContext.previousLiked?.some(liked => 
        liked.toLowerCase().includes(result.book.toLowerCase())
      )) {
        userBoost += 0.1;
      }
      
      // Boost preferred books
      if (userContext.preferredBooks?.some(book => 
        result.book.toLowerCase().includes(book.toLowerCase())
      )) {
        userBoost += 0.1;
      }
    }
    
    // Calculate total score
    // Weights: similarity (50%), topic match (40%), user boost (10%)
    const totalScore = (similarity * 0.5) + (topicMatch * 0.4) + userBoost;
    
    return {
      book: result.book,
      chapter: result.chapter,
      verse_start: result.verse_start,
      verse_end: result.verse_end,
      text: result.text,
      _similarity: similarity,
      _topicMatch: topicMatch,
      _userBoost: userBoost,
      _totalScore: totalScore
    };
  })
  .sort((a, b) => b._totalScore - a._totalScore);
}

// ============================================
// FORMAT OUTPUT
// ============================================

function formatForOutput(ranked: RankedScripture): ScriptureResult {
  return {
    book: ranked.book,
    chapter: ranked.chapter,
    verse_start: ranked.verse_start,
    verse_end: ranked.verse_end,
    text: ranked.text
  };
}
