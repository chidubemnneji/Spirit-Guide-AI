export type Archetype =

export interface PersonaProfile {
  id: string;
  userId: string;
  createdAt: string;
  updatedAt: string;
  
  archetype: Archetype;
  archetypeConfidence: number;

  trust: TrustProfile;
  currentMode: InteractionMode;
  modeHistory: ModeTransition[];
  sensitivity: SensitivityMap;
  tradition: TraditionProfile;
  evolution: EvolutionState;
  behavioralSignals: BehavioralSignals;

export interface PersonaScores {

export function createInitialPersona(
  userId: string,
  preferences: UserPreferences
): PersonaProfile {
  const archetype = determineArchetype(preferences);
  const scores = calculateInitialScores(preferences);
  
  return {
    id: crypto.randomUUID(),
    userId,
    createdAt: new Date().toISOString(),
    updatedAt: new Date().toISOString(),
    
    archetype,
    archetypeConfidence: 70,
    
    scores,
    
    trust: {
      level: 'new',
      score: 20,
      earnedThrough: [],
      lastUpdated: new Date().toISOString()
    },
    
    currentMode: ARCHETYPES[archetype].defaultMode,
    modeHistory: [],
    
    sensitivity: {
      baseline: scores.vulnerability > 70 ? 'high' : scores.vulnerability > 40 ? 'medium' : 'low',
      topics: getDefaultSensitivities(),
      detectedTriggers: [],
      volatileTopics: []
    },
    
    tradition: {
      declared: preferences.tradition || 'non-denominational',
      relationship: getInitialTraditionRelationship(preferences),
      distanceScore: preferences.intent === 'returning' ? 50 : 30,
      woundedAreas: [],
      preferTraditionalContent: preferences.intent !== 'returning',
      preferTraditionalLanguage: false
    },
    
    evolution: {
      phase: 'arriving',
      phaseStartedAt: new Date().toISOString(),
      trajectory: 'stable',
      trajectoryConfidence: 30,
      daysInApp: 0,
      daysInCurrentPhase: 0,
      comfortStagnation: false,
      stagnationDays: 0,
      growthInvitationEligible: false
    },
    
    behavioralSignals: {
      sessionsThisWeek: 0,
      averageSessionLength: 0,
      timeOfDayPattern: 'variable',
      scriptureEngagement: { offered: 0, engaged: 0, completed: 0, skipped: 0, rate: 0 },
      prayerEngagement: { offered: 0, engaged: 0, completed: 0, skipped: 0, rate: 0 },
      readingPlanEngagement: { offered: 0, engaged: 0, completed: 0, skipped: 0, rate: 0 },
      skippedContent: [],
      avoidedTopics: [],
      earlyExits: [],
      bookmarked: [],
      returnedTo: [],
      sharedDepth: 0,
      acceptedSuggestions: 0,
      declinedSuggestions: 0,
      gaveFeedback: false
    },
    
    content: {
      scriptureBias: ARCHETYPES[archetype].scriptureTopics,
      scriptureAvoid: ARCHETYPES[archetype].avoidTopics,
      preferredBooks: getPreferredBooks(archetype),
      prayerTypes: getPrayerTypes(preferences.prayerStyle),
      prayerTopics: getPrayerTopics(preferences.primaryStruggle),
      terminology: ARCHETYPES[archetype].languagePatterns,
      preferredDepth: 'surface',
      maxResponseLength: preferences.tonePreference === 'direct' ? 'brief' : 'moderate'
    },
    
    safety: {
      crisisWatch: preferences.intent === 'hard',
      crisisLevel: 'none',
      lastCrisisCheck: new Date().toISOString(),
      highVulnerability: preferences.intent === 'hard',
      authorityWoundDetected: false,
      consentRequiredForGuidance: preferences.intent === 'returning' || preferences.intent === 'hard',
      shamePatternDetected: false,
      bypassingRisk: 'low',
      dependencyRisk: 'low',
      realWorldConnectionsEncouraged: false
    }
  };
}

function determineArchetype(preferences: UserPreferences): Archetype {
  const { intent, primaryStruggle } = preferences;
  
  if (intent === 'hard') return 'wounded_seeker';
  if (intent === 'exploring') return 'curious_explorer';
  if (intent === 'returning') return 'returning_prodigal';
  if (intent === 'habit') {
    if (primaryStruggle === 'doubts') return 'struggling_saint';
    return 'eager_builder';
  }
  return 'eager_builder';
}

function calculateInitialScores(preferences: UserPreferences): PersonaScores {
  let scores: PersonaScores = {
    vulnerability: 50,
    emotionalCapacity: 50,
    intellectualEngagement: 50,
    practicalFocus: 50,
    readinessForGrowth: 50,
    readinessForDepth: 50,
    emotionalSupportNeed: 50,
    structureNeed: 50,
    cognitiveDefensiveness: 30
  };
  
  // Intent adjustments
  if (preferences.intent === 'hard') {
    scores.vulnerability += 30;
    scores.emotionalCapacity -= 20;
    scores.emotionalSupportNeed += 30;
    scores.readinessForGrowth -= 25;
  } else if (preferences.intent === 'habit') {
    scores.practicalFocus += 25;
    scores.structureNeed += 20;
    scores.readinessForGrowth += 20;
  } else if (preferences.intent === 'exploring') {
    scores.intellectualEngagement += 25;
    scores.readinessForDepth += 15;
  } else if (preferences.intent === 'returning') {
    scores.vulnerability += 15;
    scores.emotionalSupportNeed += 15;
  }
  
  // Struggle adjustments
  if (preferences.primaryStruggle === 'doubts') {
    scores.intellectualEngagement += 20;
    scores.vulnerability += 10;
  } else if (preferences.primaryStruggle === 'distant') {
    scores.emotionalSupportNeed += 25;
    scores.vulnerability += 20;
  } else if (preferences.primaryStruggle === 'consistency') {
    scores.practicalFocus += 25;
    scores.structureNeed += 20;
  }
  
  // Tone adjustments
  if (preferences.tonePreference === 'warm') {
    scores.emotionalSupportNeed += 10;
  } else if (preferences.tonePreference === 'direct') {
    scores.practicalFocus += 10;
    scores.readinessForGrowth += 10;
  }
  
  // Cap all scores
  return Object.fromEntries(
    Object.entries(scores).map(([key, value]) => [key, Math.min(100, Math.max(0, value))])
  ) as PersonaScores;
}

function getDefaultSensitivities(): TopicSensitivity[] {
  return [
    { topic: 'church', level: 'careful', source: 'default', lastUpdated: new Date().toISOString() },
    { topic: 'authority', level: 'careful', source: 'default', lastUpdated: new Date().toISOString() },
    { topic: 'sin_language', level: 'careful', source: 'default', lastUpdated: new Date().toISOString() },
    { topic: 'hell', level: 'avoid', source: 'default', lastUpdated: new Date().toISOString() },
  ];
}

function getInitialTraditionRelationship(preferences: UserPreferences): TraditionRelationship {
  if (preferences.intent === 'returning') return 'ambivalent';
  if (preferences.intent === 'exploring') return 'exploring';
  if (preferences.intent === 'hard' && preferences.primaryStruggle === 'distant') return 'wounded';
  return 'comfortable';
}

function getPreferredBooks(archetype: Archetype): string[] {
  const books: Record<Archetype, string[]> = {
    wounded_seeker: ['Psalms', 'Lamentations', 'Job', 'Isaiah'],
    eager_builder: ['Proverbs', 'James', 'Philippians', 'Psalms'],
    curious_explorer: ['Ecclesiastes', 'John', 'Luke', 'Proverbs'],
    returning_prodigal: ['Luke', 'Psalms', 'Isaiah', 'John'],
    struggling_saint: ['Psalms', 'Job', 'Mark', 'Romans']
  };
  return books[archetype];
}

function getPrayerTypes(style?: string): string[] {
  if (style === 'contemplative') return ['breath', 'traditional'];
  if (style === 'structured') return ['traditional', 'template'];
  return ['template', 'contemporary'];
}

function getPrayerTopics(struggle?: string): string[] {
  const topics: Record<string, string[]> = {
    prayer: ['learning', 'simple', 'starting'],
    consistency: ['perseverance', 'morning', 'evening'],
    doubts: ['honesty', 'questions', 'seeking'],
    distant: ['presence', 'comfort', 'connection'],
    bible: ['wisdom', 'understanding', 'guidance'],
    unsure: ['general', 'peace', 'guidance']
  };
  return topics[struggle || 'unsure'];
}

// ============================================
// SHAME DETECTION
// ============================================

export function detectShame(message: string): ShameAssessment {
  const patterns: { type: ShameType; trigger: string; evidence: string }[] = [];
  
  for (const [type, regexes] of Object.entries(SHAME_PATTERNS)) {
    for (const regex of regexes) {
      const match = message.match(regex);
      if (match) {
        patterns.push({
          type: type as ShameType,
          trigger: match[0],
          evidence: message.substring(
            Math.max(0, match.index! - 20),
            Math.min(message.length, match.index! + match[0].length + 20)
          ),
        });
        break;
      }
    }
  }
  
  if (patterns.length === 0) {
    return { detected: false, level: 'none', patterns: [], suggestedReframe: null, requiresIntervention: false };
  }
  
  let level: ShameAssessment['level'] = 'mild';
  if (patterns.length >= 3) level = 'severe';
  else if (patterns.length >= 2 || patterns.some(p => p.type === 'identity')) level = 'moderate';
  
  const primaryPattern = patterns.find(p => p.type === 'identity') || patterns[0];
  const reframes = SHAME_REFRAMES[primaryPattern.type];
  const suggestedReframe = reframes[Math.floor(Math.random() * reframes.length)];
  
  return {
    detected: true,
    level,
    patterns,
    suggestedReframe,
    requiresIntervention: level === 'severe' || primaryPattern.type === 'identity',
  };
}

export function getRandomNormalization(): string {
  return NORMALIZATION_STATEMENTS[Math.floor(Math.random() * NORMALIZATION_STATEMENTS.length)];
}

// ============================================
// VULNERABILITY & AUTHORITY WOUND DETECTION
// ============================================

const VULNERABILITY_INDICATORS = [
  /i('ve| have) never told/i,
  /this is hard to (say|admit|share)/i,
  /i('m| am) scared (to|of)/i,
  /can i be honest/i,
  /i('ve| have) been hiding/i,
  /nobody knows/i,
];

export function detectVulnerability(message: string): boolean {
  return VULNERABILITY_INDICATORS.some(pattern => pattern.test(message));
}

const AUTHORITY_WOUND_INDICATORS = [
  /don't (tell|lecture|preach at) me/i,
  /(pastors?|priests?|church(es)?|leaders?) (hurt|harmed|abused|betrayed)/i,
  /i('ve| have) been (told|taught|made to believe)/i,
  /spiritual(ly)? abuse/i,
  /i was (shamed|guilted|manipulated)/i,
];

export function detectAuthorityWound(message: string): boolean {
  return AUTHORITY_WOUND_INDICATORS.some(pattern => pattern.test(message));
}

// ============================================

export function recordTrustEvent(
  persona: PersonaProfile,
  event: string
): PersonaProfile {
  const weights: Record<string, number> = {
    returned_after_absence: 8,
    shared_vulnerable_content: 10,
    accepted_suggestion: 5,
    declined_suggestion_gracefully: 3,
    gave_positive_feedback: 4,
    gave_constructive_feedback: 6,
    stayed_during_hard_moment: 12,
    completed_reading_plan: 7,
    bookmarked_content: 2,
    long_session: 3,
  };
  
  const weight = weights[event] || 3;
  const newScore = Math.min(100, persona.trust.score + weight);
  
  let newLevel: TrustLevel = 'new';
  if (newScore >= 85) newLevel = 'deep';
  else if (newScore >= 60) newLevel = 'established';
  else if (newScore >= 25) newLevel = 'warming';
  
  return {
    ...persona,
    trust: {
      level: newLevel,
      score: newScore,
      earnedThrough: [...persona.trust.earnedThrough, {
        type: 'accepted_suggestion',
        timestamp: new Date().toISOString(),
        weight
      }].slice(-50),
      lastUpdated: new Date().toISOString()
    }
  };
}

// ============================================

export function generatePersonaPrompt(persona: PersonaProfile): string {
  const archetype = ARCHETYPES[persona.archetype];
  const mode = MODES[persona.currentMode];
  const trust = TRUST_BEHAVIORS[persona.trust.level];
  
  let prompt = `
## PERSONA CONTEXT

### Who You're Talking To
- **Archetype**: ${archetype.name}
- **Their primary needs**: ${archetype.primaryNeeds.join(', ')}
- **Their fears**: ${archetype.fears.join(', ')}

### Current State
- **Mode**: ${persona.currentMode.toUpperCase()} — ${mode.aiRole}
- **Trust Level**: ${persona.trust.level} (${persona.trust.score}/100)
- **Vulnerability**: ${describeScore(persona.scores.vulnerability)}
- **Readiness for Growth**: ${describeScore(persona.scores.readinessForGrowth)}

### Your Role Right Now
${mode.purpose}

Scripture purpose: ${mode.scripturePurpose}
Suggestions allowed: ${persona.currentMode !== 'support'}
Challenge allowed: ${trust.canChallenge}

### Language Calibration (Trust: ${persona.trust.level})
- Hedging level: ${trust.languageHedging}
- When suggesting: "${getSuggestionPrefix(persona.trust.level)}"
${trust.canChallenge ? `- When challenging: "${getChallengePrefix(persona.trust.level)}"` : '- Do not challenge yet'}

### Language to Use
${archetype.languagePatterns.use.map(p => `✓ "${p}"`).join('\n')}

### Language to Avoid
${archetype.languagePatterns.avoid.map(p => `✗ "${p}"`).join('\n')}
`;

  // Sensitivity warnings
  const avoidTopics = persona.sensitivity.topics.filter(t => t.level === 'avoid');
  const carefulTopics = persona.sensitivity.topics.filter(t => t.level === 'careful');
  
  if (avoidTopics.length > 0 || carefulTopics.length > 0) {
    prompt += `\n### Sensitivity Warnings ⚠️\n`;
    for (const topic of avoidTopics) {
      prompt += `- AVOID: "${topic.topic}"\n`;
    }
    for (const topic of carefulTopics) {
      prompt += `- CAREFUL: "${topic.topic}"\n`;
    }
  }

  // Safety flags
  if (persona.safety.consentRequiredForGuidance) {
    prompt += `\n### Consent Required ⚠️\nAlways ask before offering suggestions. Frame guidance as options.\n`;
  }

  if (persona.safety.shamePatternDetected) {
    prompt += `\n### Shame Pattern Detected ⚠️\nWatch for shame language. Reframe gently.\nRemember: "${getRandomNormalization()}"\n`;
  }

  if (persona.safety.authorityWoundDetected) {
    prompt += `\n### Authority Wound Detected ⚠️\nNever speak with directive authority. Emphasize their agency.\n`;
  }

  // Content preferences
  prompt += `
### Content Preferences
- Scripture topics to favor: ${persona.content.scriptureBias.join(', ')}
- Scripture topics to avoid: ${persona.content.scriptureAvoid.join(', ')}
- Prayer types: ${persona.content.prayerTypes.join(', ')}
- Response length: ${persona.content.maxResponseLength}
`;

  return prompt;
}

function describeScore(score: number): string {
  if (score >= 80) return 'Very high';
  if (score >= 60) return 'High';
  if (score >= 40) return 'Moderate';
  if (score >= 20) return 'Low';
  return 'Very low';
}

function getSuggestionPrefix(level: TrustLevel): string {
  switch (level) {
    case 'new': return "Some people find it helpful to... but only if that resonates.";
    case 'warming': return "One thing you might consider...";
    case 'established': return "Here's something that might help:";
    case 'deep': return "I think this could help:";
  }
}

function getChallengePrefix(level: TrustLevel): string {
  switch (level) {
    case 'new': return "";
    case 'warming': return "I wonder if...";
    case 'established': return "Can I gently push back?";
    case 'deep': return "I want to name something I'm noticing:";
  }
}
