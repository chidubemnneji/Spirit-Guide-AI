## 🧠 MEMORY EXTRACTION (REAL-TIME)

### **File:** `/server/services/memoryExtractor.js`

```javascript
const OpenAI = require('openai');
const db = require('../config/database');

const openai = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

class MemoryExtractor {
  /**
   * Extract memories from conversation
   * Called ASYNCHRONOUSLY when conversation ends
   * User doesn't wait - happens in background
   */
  async extractMemories(conversationId, userId) {
    try {
      console.log('🧠 Extracting memories for conversation:', conversationId);

      // Check if already extracted
      const existing = await db.query(
        'SELECT id FROM conversation_memories WHERE conversation_id = $1',
        [conversationId]
      );

      if (existing.rows.length > 0) {
        console.log('✅ Memories already extracted');
        return;
      }

      // Get conversation messages
      const messages = await db.query(
        `SELECT role, content, created_at
         FROM messages
         WHERE conversation_id = $1
         ORDER BY created_at ASC`,
        [conversationId]
      );

      if (messages.rows.length < 3) {
        console.log('⏭️ Conversation too short, skipping extraction');
        return;
      }

      // Build conversation text
      const conversationText = messages.rows
        .map(m => `${m.role}: ${m.content}`)
        .join('\n\n');

      // Extract with GPT-4o-mini (cheap and fast)
      const response = await openai.chat.completions.create({
        model: 'gpt-4o-mini',
        messages: [
          {
            role: 'system',
            content: `You are analyzing a spiritual conversation to extract key insights.

Extract and return JSON with these fields:
{
  "key_topics": ["topic1", "topic2"],           // Max 5, be SPECIFIC ("work stress" not "stress")
  "emotional_patterns": ["emotion1", "emotion2"], // Max 5
  "struggles": ["struggle1", "struggle2"],       // Max 5, be DETAILED
  "growth_moments": ["moment1", "moment2"],      // Max 3, actual breakthroughs
  "recurring_questions": ["question1"],          // Max 3, deep questions they're wrestling with
  "faith_stage_indicators": "beginner/growing/mature", // Based on conversation depth
  "summary": "2-sentence summary of what this conversation was about"
}

BE SPECIFIC. Examples:
Good: "Struggling to set boundaries with coworkers"
Bad: "Work problems"

Good: "Wondering if God still hears prayers after past failures"
Bad: "Doubting"

Extract meaningful, actionable insights.`
          },
          {
            role: 'user',
            content: `Extract insights from this conversation:\n\n${conversationText}`
          }
        ],
        response_format: { type: 'json_object' },
        temperature: 0.3, // Low temp for consistent extraction
      });

      const insights = JSON.parse(response.choices[0].message.content);

      // Save to database
      await db.query(
        `INSERT INTO conversation_memories (
          user_id,
          conversation_id,
          key_topics,
          emotional_patterns,
          struggles,
          growth_moments,
          recurring_questions,
          faith_stage_indicators,
          summary
        ) VALUES ($1, $2, $3, $4, $5, $6, $7, $8, $9)`,
        [
          userId,
          conversationId,
          insights.key_topics || [],
          insights.emotional_patterns || [],
          insights.struggles || [],
          insights.growth_moments || [],
          insights.recurring_questions || [],
          insights.faith_stage_indicators || 'beginner',
          insights.summary || ''
        ]
      );

      console.log('✅ Memories extracted and saved:', {
        topics: insights.key_topics?.length || 0,
        struggles: insights.struggles?.length || 0,
        growth: insights.growth_moments?.length || 0
      });

    } catch (error) {
      console.error('❌ Memory extraction error:', error);
      // Don't throw - this is async, don't break user flow
    }
  }

  /**
   * Re-extract memories (for fixing bad extractions)
   */
  async reExtractMemories(conversationId, userId) {
    try {
      // Delete existing
      await db.query(
        'DELETE FROM conversation_memories WHERE conversation_id = $1',
        [conversationId]
      );

      // Extract fresh
      await this.extractMemories(conversationId, userId);
      
      console.log('✅ Memories re-extracted for:', conversationId);
    } catch (error) {
      console.error('Re-extract error:', error);
      throw error;
    }
  }
}

module.exports = new MemoryExtractor();
```

### **Update Chat Controller to Extract Memories**

**File:** `/server/controllers/chatController.js`

```javascript
const memoryExtractor = require('../services/memoryExtractor');

/**
 * End conversation and extract memories
 * USER DOESN'T WAIT - memories extracted in background
 */
router.post('/conversations/:id/end', authenticateToken, async (req, res) => {
  try {
    const conversationId = req.params.id;
    const userId = req.user.id;

    // Mark conversation as ended
    await db.query(
      `UPDATE conversations 
       SET ended_at = NOW() 
       WHERE id = $1 AND user_id = $2`,
      [conversationId, userId]
    );

    // ✅ CRITICAL: Extract memories ASYNCHRONOUSLY
    // No "await" here - user doesn't wait
    // Happens in background (2-3 seconds)
    memoryExtractor.extractMemories(conversationId, userId)
      .catch(err => console.error('Background memory extraction failed:', err));

    // ✅ User gets instant response (0.1 seconds)
    res.json({ 
      success: true, 
      message: 'Conversation ended' 
    });

  } catch (error) {
    console.error('End conversation error:', error);
    res.status(500).json({ 
      success: false, 
      error: 'Failed to end conversation' 
    });
  }
});
```

```javascript
/**
 * Get devotional context for AI conversation
 */
async getDevotionalContext(userId) {
  try {
    // Get today's devotional if user has one
    const todayDevotional = await db.query(
      `SELECT d.*, dda.is_completed
       FROM daily_devotional_assignments dda
       JOIN devotionals d ON dda.devotional_id = d.id
       WHERE dda.user_id = $1 
       AND dda.assigned_date = CURRENT_DATE`,
      [userId]
    );

    if (todayDevotional.rows.length === 0) {
      return null;
    }

    const devotional = todayDevotional.rows[0];

    return `
═══════════════════════════════════════════════════════════
TODAY'S DEVOTIONAL CONTEXT
═══════════════════════════════════════════════════════════

The user has been reading today's devotional:

Key themes: ${devotional.themes?.join(', ')}

Today's practice: ${devotional.todays_practice}

Status: ${devotional.is_completed ? 'Completed' : 'In progress'}

IMPORTANT INSTRUCTIONS:
- If the user mentions "today's devotional" or "Begin Reflection" button led here, you have this context
- Reference the scripture and themes naturally in your responses
- Don't assume they completed it unless they say so

- Ask questions that help them apply it to their life
    `;

  } catch (error) {

  }
}
```
