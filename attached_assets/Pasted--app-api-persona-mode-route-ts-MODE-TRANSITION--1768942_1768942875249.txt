// app/api/persona/mode/route.ts
// ============================================
// MODE TRANSITION API
// Change between support/formation/learning modes
// ============================================

import { NextRequest, NextResponse } from 'next/server';
import { getPersona, updateMode } from '@/lib/supabase/persona-service';
import { MODES } from '@/lib/persona/complete-persona-system';

type Mode = 'support' | 'formation' | 'learning';
type Trigger = 'user_request' | 'stability_detected' | 'gentle_invitation' | 'crisis_regression';

// ============================================
// GET - Get current mode info
// ============================================

export async function GET(request: NextRequest) {
  try {
    const { searchParams } = new URL(request.url);
    const userId = searchParams.get('userId');
    
    if (!userId) {
      return NextResponse.json(
        { error: 'Missing userId parameter' },
        { status: 400 }
      );
    }
    
    const persona = await getPersona(userId);
    
    if (!persona) {
      return NextResponse.json(
        { error: 'Persona not found' },
        { status: 404 }
      );
    }
    
    const currentMode = MODES[persona.currentMode];
    
    return NextResponse.json({
      currentMode: persona.currentMode,
      modeInfo: {
        name: currentMode.name,
        purpose: currentMode.purpose,
        aiRole: currentMode.aiRole,
        scripturePurpose: currentMode.scripturePurpose
      },
      canTransitionTo: currentMode.canTransitionTo,
      recentHistory: persona.modeHistory.slice(-5)
    });
    
  } catch (error) {
    console.error('Get mode error:', error);
    return NextResponse.json(
      { error: 'Failed to get mode' },
      { status: 500 }
    );
  }
}

// ============================================
// POST - Request mode transition
// ============================================

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { userId, newMode, trigger, userConsented } = body;
    
    // Validation
    if (!userId || !newMode) {
      return NextResponse.json(
        { error: 'Missing userId or newMode' },
        { status: 400 }
      );
    }
    
    if (!['support', 'formation', 'learning'].includes(newMode)) {
      return NextResponse.json(
        { error: 'Invalid mode. Must be: support, formation, or learning' },
        { status: 400 }
      );
    }
    
    const validTriggers: Trigger[] = ['user_request', 'stability_detected', 'gentle_invitation', 'crisis_regression'];
    const actualTrigger: Trigger = validTriggers.includes(trigger) ? trigger : 'user_request';
    
    // Get current persona
    const persona = await getPersona(userId);
    if (!persona) {
      return NextResponse.json(
        { error: 'Persona not found' },
        { status: 404 }
      );
    }
    
    // Check if transition is allowed
    const currentModeConfig = MODES[persona.currentMode];
    if (!currentModeConfig.canTransitionTo.includes(newMode as Mode)) {
      // Special case: support → learning must go through formation
      if (persona.currentMode === 'support' && newMode === 'learning') {
        return NextResponse.json({
          allowed: false,
          error: 'Cannot transition directly from support to learning',
          suggestion: 'Consider transitioning to formation first',
          consentPrompt: "Before we dive into deeper questions, would it help to build some grounding practices first?"
        }, { status: 400 });
      }
      
      return NextResponse.json({
        allowed: false,
        error: `Cannot transition from ${persona.currentMode} to ${newMode}`
      }, { status: 400 });
    }
    
    // Check consent requirements
    if (currentModeConfig.transitionRequires === 'consent' && !userConsented) {
      return NextResponse.json({
        allowed: false,
        requiresConsent: true,
        consentPrompt: getConsentPrompt(persona.currentMode, newMode as Mode)
      });
    }
    
    // Perform transition
    const success = await updateMode(
      userId,
      newMode as Mode,
      actualTrigger,
      userConsented ?? true
    );
    
    if (!success) {
      return NextResponse.json(
        { error: 'Failed to update mode' },
        { status: 500 }
      );
    }
    
    const newModeConfig = MODES[newMode as Mode];
    
    return NextResponse.json({
      success: true,
      previousMode: persona.currentMode,
      newMode,
      modeInfo: {
        name: newModeConfig.name,
        purpose: newModeConfig.purpose,
        aiRole: newModeConfig.aiRole
      },
      transitionMessage: getTransitionMessage(persona.currentMode, newMode as Mode)
    });
    
  } catch (error) {
    console.error('Mode transition error:', error);
    return NextResponse.json(
      { error: 'Failed to transition mode' },
      { status: 500 }
    );
  }
}

// ============================================
// HELPERS
// ============================================

function getConsentPrompt(from: Mode, to: Mode): string {
  const prompts: Record<string, string> = {
    'support→formation': "You've been carrying a lot. I want to make sure — are you in a place where some practical suggestions might help? Or would you rather we just keep talking?",
    'support→learning': "That's a great question to explore. Before we go there, I want to check — how are you feeling right now?",
    'formation→learning': "Would you like to explore the 'why' behind what we've been practicing?",
    'formation→support': "It sounds like you might need to pause on the 'doing' for a moment. Want to just talk?",
    'learning→support': "This is heavy. Want to step back from the ideas for a bit?",
    'learning→formation': "Want to try putting some of this into practice?"
  };
  
  return prompts[`${from}→${to}`] || "Would you like to shift gears?";
}

function getTransitionMessage(from: Mode, to: Mode): string {
  const messages: Record<string, string> = {
    'support→formation': "Okay, let me share something you might try...",
    'support→learning': "Let's explore that together...",
    'formation→support': "Let's pause on the 'doing' for a moment. How are you actually feeling?",
    'formation→learning': "Good question. Let me share some context...",
    'learning→support': "That's heavy. Let's step back from the ideas for a second.",
    'learning→formation': "Want to try putting some of this into practice?"
  };
  
  return messages[`${from}→${to}`] || "";
}
