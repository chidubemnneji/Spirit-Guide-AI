# NEXT TASK: Build the Chat Interface

## What This Is

The chat interface where users have conversations with their AI spiritual companion. This is where all your work comes together:
- System Prompt Builder generates personalized responses
- User's persona data shapes the AI's tone
- Pastoral response phases (acknowledge → console → reflect → recommend) are visible
- Conversation history is saved

```typescript
import MessageInput from './MessageInput';
import TypingIndicator from './TypingIndicator';

const ChatInterface: React.FC = () => {
  const { user } = useAuth();
  const {
    currentConversation,
    messages,
```

```typescript
import React from 'react';
import { View, Text, StyleSheet } from 'react-native';
import { Message } from '../../types/chat.types';

interface MessageBubbleProps {
  message: Message;
}

const MessageBubble: React.FC<MessageBubbleProps> = ({ message }) => {
  const isUser = message.role === 'user';

  return (
    <View style={[
      styles.container,
      isUser ? styles.userContainer : styles.assistantContainer
    ]}>
      <View style={[
        styles.bubble,
        isUser ? styles.userBubble : styles.assistantBubble
      ]}>
        <Text style={[
          styles.text,
          isUser ? styles.userText : styles.assistantText
        ]}>
          {message.content}
        </Text>
        <Text style={[
          styles.timestamp,
          isUser ? styles.userTimestamp : styles.assistantTimestamp
        ]}>
          {formatTime(message.createdAt)}
        </Text>
      </View>
    </View>
  );
};

const formatTime = (date: Date | string): string => {
  const d = typeof date === 'string' ? new Date(date) : date;
  const now = new Date();
  const diffMs = now.getTime() - d.getTime();
  const diffMins = Math.floor(diffMs / 60000);

  if (diffMins < 1) return 'Just now';
  if (diffMins < 60) return `${diffMins}m ago`;
  
  const diffHours = Math.floor(diffMins / 60);
  if (diffHours < 24) return `${diffHours}h ago`;
  
  return d.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
};

const styles = StyleSheet.create({

  },
```

```typescript
import React, { useEffect, useRef } from 'react';
import { View, Text, StyleSheet, Animated } from 'react-native';

const TypingIndicator: React.FC = () => {
  const dot1 = useRef(new Animated.Value(0)).current;
  const dot2 = useRef(new Animated.Value(0)).current;
  const dot3 = useRef(new Animated.Value(0)).current;

  useEffect(() => {
    const animate = (dot: Animated.Value, delay: number) => {
      Animated.loop(
        Animated.sequence([
          Animated.delay(delay),
          Animated.timing(dot, {
            toValue: 1,
            duration: 400,
            useNativeDriver: true,
          }),
          Animated.timing(dot, {
            toValue: 0,
            duration: 400,
            useNativeDriver: true,
          }),
        ])
      ).start();
    };

    animate(dot1, 0);
    animate(dot2, 200);
    animate(dot3, 400);
  }, []);

  return (
    <View style={styles.container}>
      <View style={styles.bubble}>
        <Text style={styles.label}>AI is thinking</Text>
        <View style={styles.dots}>
          <Animated.View
            style={[
              styles.dot,
              {
                opacity: dot1.interpolate({
                  inputRange: [0, 1],
                  outputRange: [0.3, 1],
                }),
              },

              styles.dot,
              {
```

## BACKEND: Enhanced Chat Controller

**File:** `/server/controllers/chatController.js`

```javascript
const db = require('../services/databaseService');
const anthropic = require('../services/anthropicService');
const systemPromptBuilder = require('../services/systemPromptBuilder');

/**
 * Create new conversation
 */
exports.createConversation = async (req, res) => {
  try {
    const userId = req.user.id;

    const result = await db.query(
      `INSERT INTO conversations (user_id, title, message_count, current_phase)
       VALUES ($1, $2, $3, $4)
       RETURNING id, title, created_at`,
      [userId, 'New Conversation', 0, 'acknowledgment']
    );

    res.status(201).json({
      success: true,
      data: result.rows[0]
    });
  } catch (error) {
    console.error('Create conversation error:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to create conversation'
    });
  }
};

/**
 * Get conversation by ID
 */
exports.getConversation = async (req, res) => {
  try {
    const { conversationId } = req.params;
    const userId = req.user.id;

    // Get conversation
    const conversationResult = await db.query(
      `SELECT * FROM conversations 
       WHERE id = $1 AND user_id = $2`,
      [conversationId, userId]
    );

    if (conversationResult.rows.length === 0) {
      return res.status(404).json({
        success: false,
        error: 'Conversation not found'
      });
    }

    // Get messages
    const messagesResult = await db.query(
      `SELECT * FROM messages 
       WHERE conversation_id = $1 
       ORDER BY created_at ASC`,
      [conversationId]
    );

    res.json({
      success: true,
      data: {
        conversation: conversationResult.rows[0],
        messages: messagesResult.rows
      }
    });
  } catch (error) {
    console.error('Get conversation error:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to get conversation'
    });
  }
};

/**
 * Send message and get AI response
 */
exports.sendMessage = async (req, res) => {
  try {
    const { conversationId } = req.params;
    const { content } = req.body;
    const userId = req.user.id;

    // Verify conversation belongs to user
    const conversationResult = await db.query(
      `SELECT message_count FROM conversations 
       WHERE id = $1 AND user_id = $2`,
      [conversationId, userId]
    );

    if (conversationResult.rows.length === 0) {
      return res.status(404).json({
        success: false,
        error: 'Conversation not found'
      });
    }

    const messageCount = conversationResult.rows[0].message_count;

    // Get user persona
    const personaResult = await db.query(
      `SELECT 
        u.name as user_name,
        up.primary_struggle,
        up.depth_layer_responses,
        up.daily_rhythm,
        up.past_connection_moment,
        up.connection_recency,
        up.peak_energy_time,
        up.obstacles,
        up.transformation_goals,
        up.primary_persona,
        up.persona_modifiers
      FROM user_personas up
      JOIN users u ON up.user_id = u.id
      WHERE up.user_id = $1`,
      [userId]
    );

    if (personaResult.rows.length === 0) {
      return res.status(400).json({
        success: false,
        error: 'User persona not found. Please complete onboarding.'
      });
    }

    const userPersona = personaResult.rows[0];

    // Build system prompt with current phase
    const systemPrompt = systemPromptBuilder.buildPrompt(userPersona, messageCount);

    // Get conversation history
    const historyResult = await db.query(
      `SELECT role, content 
       FROM messages 
       WHERE conversation_id = $1 
       ORDER BY created_at ASC`,
      [conversationId]
    );

    const messages = historyResult.rows.map(msg => ({
      role: msg.role,
      content: msg.content
    }));

    // Add new user message
    messages.push({
      role: 'user',
      content: content
    });

    // Call Claude API
    const response = await anthropic.messages.create({
      model: 'claude-sonnet-4-20250514',
      system: systemPrompt,
      messages: messages,
      max_tokens: 1000
    });

    const aiResponse = response.content[0].text;

    // Save user message
    const userMessageResult = await db.query(
      `INSERT INTO messages (conversation_id, role, content)
       VALUES ($1, $2, $3)
       RETURNING *`,
      [conversationId, 'user', content]
    );

    // Save AI message
    const aiMessageResult = await db.query(
      `INSERT INTO messages (conversation_id, role, content)
       VALUES ($1, $2, $3)
       RETURNING *`,
      [conversationId, 'assistant', aiResponse]
    );

    // Update conversation
    const newMessageCount = messageCount + 2;
    const newPhase = determinePhase(newMessageCount);

    await db.query(
      `UPDATE conversations 
       SET message_count = $1,
           current_phase = $2,
           updated_at = NOW()
       WHERE id = $3`,
      [newMessageCount, newPhase, conversationId]
    );

    // Update conversation title if it's the first message
    if (messageCount === 0) {
      const title = content.substring(0, 50) + (content.length > 50 ? '...' : '');
      await db.query(
        'UPDATE conversations SET title = $1 WHERE id = $2',
        [title, conversationId]
      );
    }

    res.json({
      success: true,
      data: {
        userMessage: userMessageResult.rows[0],
        aiMessage: aiMessageResult.rows[0]
      }
    });

  } catch (error) {
    console.error('Send message error:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to send message'
    });
  }
};

/**
 * Get user's conversations
 */
exports.getUserConversations = async (req, res) => {
  try {
    const userId = req.user.id;

    const result = await db.query(
      `SELECT id, title, message_count, current_phase, created_at, updated_at
       FROM conversations
       WHERE user_id = $1
       ORDER BY updated_at DESC
       LIMIT 20`,
      [userId]
    );

    res.json({
      success: true,
      data: result.rows
    });
  } catch (error) {
    console.error('Get conversations error:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to get conversations'
    });
  }
};

/**
 * Determine conversation phase based on message count
 */
function determinePhase(messageCount) {
  if (messageCount <= 1) return 'acknowledgment';
  if (messageCount === 2) return 'consolation';
  if (messageCount === 3) return 'reflection';
  return 'recommendation';
}
```
