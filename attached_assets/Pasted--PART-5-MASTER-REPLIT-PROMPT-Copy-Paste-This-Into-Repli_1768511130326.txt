```sql
-- Conversation topics across all chats
CREATE TABLE conversation_topics (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  topic VARCHAR(100),
  first_mentioned TIMESTAMP DEFAULT NOW(),
  last_mentioned TIMESTAMP DEFAULT NOW(),
  mention_count INTEGER DEFAULT 1,
  sentiment VARCHAR(20),
  notes TEXT,
  UNIQUE(user_id, topic)
);

CREATE INDEX idx_topics_user ON conversation_topics(user_id, last_mentioned DESC);

-- Memorable moments
CREATE TABLE memorable_moments (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  conversation_id UUID REFERENCES conversations(id),
  moment_type VARCHAR(50),
  summary TEXT,
  emotional_state VARCHAR(50),
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_moments_user ON memorable_moments(user_id, created_at DESC);

-- Spiritual practices log
CREATE TABLE practices_log (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  practice_type VARCHAR(100),
  suggested_by_ai BOOLEAN DEFAULT false,
  completed BOOLEAN DEFAULT false,
  helpful_rating INTEGER,
  reflection TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_practices_user ON practices_log(user_id, created_at DESC);

-- Emotional check-ins
CREATE TABLE emotional_checkins (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  emotional_state VARCHAR(50),
  intensity INTEGER,
  context TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_emotions_user ON emotional_checkins(user_id, created_at DESC);

-- Pattern analysis
CREATE TABLE pattern_analysis (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  analysis_data JSONB,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_patterns_user ON pattern_analysis(user_id, created_at DESC);

-- Crisis alerts
CREATE TABLE crisis_alerts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id UUID REFERENCES users(id) ON DELETE CASCADE,
  crisis_level VARCHAR(20),
  indicators TEXT[],
  message_excerpt TEXT,
  reviewed BOOLEAN DEFAULT false,
  created_at TIMESTAMP DEFAULT NOW()
);

CREATE INDEX idx_crisis_user ON crisis_alerts(user_id, created_at DESC);
```
