      const practices = await db.query(
        `SELECT practice_type, completed, helpful_rating, created_at

         WHERE user_id = $1

        `SELECT analysis_data, created_at 
         FROM pattern_analysis 
         WHERE user_id = $1 
         ORDER BY created_at DESC 
         LIMIT 1`,
        [userId]
      );

      // Re-analyze if more than 7 days old or doesn't exist
      if (!lastAnalysis.rows[0] || this.isStale(lastAnalysis.rows[0].created_at, 7)) {
        const patterns = await this.analyzePatterns(userId);
        
        await db.query(
          `INSERT INTO pattern_analysis (user_id, analysis_data)
           VALUES ($1, $2)`,
          [userId, JSON.stringify(patterns)]

      }

      return JSON.parse(lastAnalysis.rows[0].analysis_data);

    } catch (error) {
      console.error('Get patterns error:', error);
      return { patterns: [] };
    }
  }

  isStale(date, days) {
    const diffTime = Math.abs(new Date() - new Date(date));

const db = require('./databaseService');

class ProgressTracker {
  /**
   * Detect milestones
   */
  async detectMilestones(userId, userName) {
    const milestones = [];

    try {
      // Check conversation streak
      const streak = await this.getConversationStreak(userId);
      if (streak === 7) {
        milestones.push({
          type: 'streak',
          title: '7 Days of Showing Up',
          message: `${userName}, you've been here 7 days in a row. That's you choosing to show up for your faith journey every single day this week. That's not small.`
        });
      } else if (streak === 30) {
        milestones.push({
          type: 'streak',
          title: '30 Days Strong',
          message: `A whole month, ${userName}. You've shown up 30 days in a row. Whatever you're doing, it's working.`
        });
      }

      // Check emotional improvement
      const emotionalTrend = await this.getEmotionalTrend(userId);
      if (emotionalTrend === 'improving') {
        milestones.push({
          type: 'emotional',
          title: 'Growing Peace',
          message: 'I\'ve noticed something: your emotional check-ins over the past 2 weeks show a shift toward more peace. Whatever you\'re doing, keep going.'
        });
      }

      // Check practice completion
      const practicesCompleted = await this.getPracticeCompletionCount(userId);
      if (practicesCompleted === 5) {
        milestones.push({
          type: 'practice',
          title: '5 Practices Completed',
          message: 'You\'ve tried 5 different spiritual practices. That takes courageвЂ”especially when you\'re figuring out what works. You\'re exploring.'
        });
      } else if (practicesCompleted === 10) {
        milestones.push({
          type: 'practice',
          title: '10 Practices',
          message: 'Ten practices. You\'re not just talking about faithвЂ”you\'re actually doing it. That matters.'
        });
      }

      // Check topic resolution
      const resolvedTopics = await this.getResolvedTopics(userId);
      if (resolvedTopics.length > 0) {
        resolvedTopics.forEach(topic => {
          milestones.push({
            type: 'breakthrough',
            title: `Progress on ${topic}`,
            message: `Remember when you first brought up ${topic}? Look at where you are now. That's growth.`
          });
        });
      }

      return milestones;

    } catch (error) {
      console.error('Milestone detection error:', error);
      return [];
    }
  }

  async getConversationStreak(userId) {
    const result = await db.query(
      `WITH RECURSIVE date_series AS (
        SELECT CURRENT_DATE - INTERVAL '30 days' as date
        UNION ALL
        SELECT date + INTERVAL '1 day'
        FROM date_series
        WHERE date < CURRENT_DATE
      ),
      user_dates AS (
        SELECT DISTINCT DATE(created_at) as conversation_date
        FROM conversations
        WHERE user_id = $1
        AND created_at >= CURRENT_DATE - INTERVAL '30 days'
      ),
      streak AS (
        SELECT 
          ds.date,
          CASE WHEN ud.conversation_date IS NOT NULL THEN 1 ELSE 0 END as had_conversation,
          ROW_NUMBER() OVER (ORDER BY ds.date DESC) as rn
        FROM date_series ds
        LEFT JOIN user_dates ud ON ds.date = ud.conversation_date
        WHERE ds.date <= CURRENT_DATE
        ORDER BY ds.date DESC
      )
      SELECT COUNT(*) as streak_length
      FROM (
        SELECT had_conversation, 
               SUM(CASE WHEN had_conversation = 0 THEN 1 ELSE 0 END) 
                 OVER (ORDER BY rn) as gap_group
        FROM streak
        WHERE rn <= 30
      ) t
      WHERE gap_group = 0 AND had_conversation = 1`,
      [userId]
    );

    return result.rows[0]?.streak_length || 0;
  }

  async getEmotionalTrend(userId) {
    const recentEmotions = await db.query(
      `SELECT emotional_state, intensity, created_at
       FROM emotional_checkins
       WHERE user_id = $1
       AND created_at >= NOW() - INTERVAL '14 days'
       ORDER BY created_at ASC`,
      [userId]
    );

    if (recentEmotions.rows.length < 4) return 'insufficient_data';

    // Compare first half to second half
    const midpoint = Math.floor(recentEmotions.rows.length / 2);
    const firstHalf = recentEmotions.rows.slice(0, midpoint);
    const secondHalf = recentEmotions.rows.slice(midpoint);

    const avgFirst = firstHalf.reduce((sum, e) => sum + e.intensity, 0) / firstHalf.length;
    const avgSecond = secondHalf.reduce((sum, e) => sum + e.intensity, 0) / secondHalf.length;

    // Lower intensity = better (less struggle)
    if (avgSecond < avgFirst - 1.5) return 'improving';
    if (avgSecond > avgFirst + 1.5) return 'worsening';
    return 'stable';
  }

  async getPracticeCompletionCount(userId) {
    const result = await db.query(
      `SELECT COUNT(*) as count
       FROM practices_log
       WHERE user_id = $1 AND completed = true`,
      [userId]
    );
    return parseInt(result.rows[0]?.count || 0);
  }

  async getResolvedTopics(userId) {
    const result = await db.query(
      `SELECT topic
       FROM conversation_topics
       WHERE user_id = $1
       AND sentiment = 'resolved'
       AND last_mentioned >= NOW() - INTERVAL '30 days'`,
      [userId]
    );
    return result.rows.map(r => r.topic);
  }

  /**
   * Format milestone for system prompt
   */
  formatMilestoneForPrompt(milestone) {
    if (!milestone) return '';

    return `
рџЋЉ CELEBRATION OPPORTUNITY:
The user has reached a milestone: ${milestone.title}

IF it flows naturally in this conversation, you can acknowledge:
"${milestone.message}"

But only if appropriate - don't force it.
`;
  }
}

      "instructions": "Step by step guidance"
    }
  ]

      const response = await anthropic.messages.create({
        model: 'claude-sonnet-4-20250514',
        messages: [{ role: 'user', content: recommendationPrompt }],
        max_tokens: 1500
      });

      const cleaned = response.content[0].text.replace(/```json|```/g, '').trim();
      return JSON.parse(cleaned);

    } catch (error) {
      console.error('Recommendation generation error:', error);
      return { recommendations: [] };
    }
  }

  async extractConstraints(message) {
    // Simple keyword extraction for constraints
    const constraints = {
      available_time: 'unknown',
      energy_level: 'unknown',
      barriers: []
    };

    const lowerMessage = message.toLowerCase();

    // Time constraints
    if (lowerMessage.includes('no time') || lowerMessage.includes('busy')) {
      constraints.available_time = 'very limited';
    } else if (lowerMessage.includes('few minutes') || lowerMessage.includes('quick')) {
      constraints.available_time = 'under 5 minutes';
    }

    // Energy level
    if (lowerMessage.includes('exhausted') || lowerMessage.includes('tired') || lowerMessage.includes('drained')) {
      constraints.energy_level = 'very low';
    } else if (lowerMessage.includes('overwhelmed')) {
      constraints.energy_level = 'low';
    }

    return constraints;
  }

  async getSuccessfulPractices(userId) {
    const result = await db.query(
      `SELECT practice_type, helpful_rating, COUNT(*) as times_completed
       FROM practices_log
       WHERE user_id = $1
       AND completed = true
       AND helpful_rating >= 4
       GROUP BY practice_type, helpful_rating
       ORDER BY helpful_rating DESC, times_completed DESC
       LIMIT 3`,
      [userId]
    );

    return result.rows;
  }
}

module.exports = new RecommendationEngine();

      if (daysSince <= 2) {
        bridgeContext += `Since it's been just ${daysSince} day${daysSince === 1 ? '' : 's'}, you might say:\n`;
        bridgeContext += `"Good to see you again, ${userName}. How have things been since we talked about ${last.title}?"\n\n`;
      } else if (daysSince <= 7) {
        bridgeContext += `It's been about a week, so you might say:\n`;
        bridgeContext += `"Hey ${userName}, it's been a few days. I've been thinking about what you shared last time..."\n\n`;
      } else {
        bridgeContext += `It's been a while (${daysSince} days), so you might say:\n`;
        bridgeContext += `"${userName}, it's good to hear from you. It's been a little whileвЂ”how have you been?"\n\n`;
      }

      bridgeContext += 'Only reference past conversations if it feels natural.\n';
      bridgeContext += 'If the user jumps straight into something new, follow their lead.\n';

      return bridgeContext;

    } catch (error) {
      console.error('Conversation bridge error:', error);
      return '';
    }
  }
}

module.exports = new ConversationBridge();
