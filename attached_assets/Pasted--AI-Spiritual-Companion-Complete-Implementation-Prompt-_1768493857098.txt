# AI Spiritual Companion - Complete Implementation Prompt for Replit

## Project Overview
Build a spiritual companion app with a multi-phase onboarding flow that creates deep user personas, then delivers personalized AI responses based on 8 distinct persona types.

---

## Tech Stack Requirements
- **Frontend:** React (with Tailwind CSS for styling)
- **Backend:** Node.js/Express
- **Database:** PostgreSQL or MongoDB (your choice)
- **AI Integration:** Anthropic Claude API (Claude Sonnet 4)
- **State Management:** React Context or Redux
- **Routing:** React Router

---

## Database Schema

### Users Table
```sql
CREATE TABLE users (
  id UUID PRIMARY KEY,
  email VARCHAR(255) UNIQUE,
  name VARCHAR(255),
  created_at TIMESTAMP DEFAULT NOW(),
  last_active TIMESTAMP
);
```

### User Personas Table
```sql
CREATE TABLE user_personas (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  
  -- Phase 1: Emotional Entry
  primary_struggle VARCHAR(100),
  
  -- Phase 2: Depth Layer (JSON for flexibility)
  depth_layer_responses JSONB,
  
  -- Phase 3: Behavioral Reality
  daily_rhythm TEXT[],
  past_connection_moment VARCHAR(100),
  connection_recency VARCHAR(50),
  peak_energy_time VARCHAR(50),
  obstacles TEXT[],
  
  -- Phase 4: Hope Layer
  transformation_goals TEXT[],
  
  -- Generated Persona
  primary_persona VARCHAR(50),
  persona_modifiers TEXT[],
  
  created_at TIMESTAMP DEFAULT NOW(),
  updated_at TIMESTAMP DEFAULT NOW()
);
```

### Conversations Table
```sql
CREATE TABLE conversations (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  created_at TIMESTAMP DEFAULT NOW()
);
```

### Messages Table
```sql
CREATE TABLE messages (
  id UUID PRIMARY KEY,
  conversation_id UUID REFERENCES conversations(id),
  role VARCHAR(20), -- 'user' or 'assistant'
  content TEXT,
  created_at TIMESTAMP DEFAULT NOW()
);
```

### Frontend Component: Phase1.jsx

```jsx
import React, { useState } from 'react';
import { useOnboarding } from '../../context/OnboardingContext';

const Phase1 = ({ onNext }) => {
  const { updateOnboarding } = useOnboarding();
  const [selected, setSelected] = useState(null);

  const options = [
    {
      id: 'distant_from_god',
      icon: '🌫️',
      text: 'I feel distant from God - prayer feels empty'
    },
    {
      id: 'wrestling_doubts',
      icon: '❓',
      text: "I'm wrestling with doubts I can't shake"
    },
    {
      id: 'feel_alone',
      icon: '🚶',
      text: 'I feel alone in my faith journey'
    },
    {
      id: 'guilt_shame',
      icon: '💔',
      text: "I'm carrying guilt or shame I can't let go of"
    },
    {
      id: 'life_overwhelming',
      icon: '⚡',
      text: 'Life is overwhelming and my faith is slipping'
    },
    {
      id: 'new_to_faith',
      icon: '🌱',
      text: "I'm new to faith and don't know where to start"
    }
  ];

  const handleSelect = (optionId) => {
    setSelected(optionId);
  };

  const handleNext = () => {
    if (selected) {
      updateOnboarding({ primary_struggle: selected });
      onNext(selected); // Pass to parent to route to correct Phase 2
    }
  };

  return (
    <div className="max-w-2xl mx-auto p-6">
      <h1 className="text-3xl font-bold text-center mb-4">
        Before we begin, we want to understand where you are right now.
      </h1>
      <p className="text-lg text-center text-gray-600 mb-8">
        Which of these feels most true for you today?
      </p>

      <div className="space-y-4">
        {options.map((option) => (
          <button
            key={option.id}
            onClick={() => handleSelect(option.id)}
            className={`w-full p-6 text-left rounded-lg border-2 transition-all ${
              selected === option.id
                ? 'border-blue-500 bg-blue-50'
                : 'border-gray-200 hover:border-gray-300'
            }`}
          >
            <div className="flex items-center gap-4">
              <span className="text-4xl">{option.icon}</span>
              <span className="text-lg">{option.text}</span>
            </div>
          </button>
        ))}
      </div>

      <button
        onClick={handleNext}
        disabled={!selected}
        className={`w-full mt-8 py-4 px-6 rounded-lg font-semibold text-white transition-all ${
          selected
            ? 'bg-blue-600 hover:bg-blue-700'
            : 'bg-gray-300 cursor-not-allowed'
        }`}
      >
        Continue
      </button>
    </div>
  );
};
```

```jsx
import React, { useState } from 'react';
import { useOnboarding } from '../../context/OnboardingContext';

const Phase2A_DistantFromGod = ({ onNext }) => {
  const { updateOnboarding } = useOnboarding();
  const [step, setStep] = useState(1);
  const [answers, setAnswers] = useState({
    prayer_experience: [],
    past_connection: null
  });

  const question1Options = [
    { id: 'empty_room', text: 'Like talking to an empty room' },
    { id: 'rushed_mechanical', text: 'Rushed and mechanical, just going through motions' },
    { id: 'fades_quickly', text: 'I feel something but it fades quickly' },
    { id: 'dont_know_how', text: "I want to connect but don't know how" }
  ];

  const question2Options = [
    { id: 'far_away_now', text: 'Yes, but it feels far away now' },
    { id: 'not_sure_ever', text: "I'm not sure I ever have" },
    { id: 'something_changed', text: 'I used to, but something changed' },
    { id: 'catch_glimpses', text: "I catch glimpses but can't hold onto them" }
  ];

  const handleMultiSelect = (optionId) => {
    setAnswers((prev) => ({
      ...prev,
      prayer_experience: prev.prayer_experience.includes(optionId)
        ? prev.prayer_experience.filter((id) => id !== optionId)
        : [...prev.prayer_experience, optionId]

    setAnswers((prev) => ({
      ...prev,
      past_connection: optionId
    }));
  };

  const handleNext = () => {
    if (step === 1 && answers.prayer_experience.length > 0) {
      setStep(2);
    } else if (step === 2 && answers.past_connection) {
      updateOnboarding({ depth_layer: answers });
      onNext();
    }
  };

  return (
    <div className="max-w-2xl mx-auto p-6">
      {step === 1 && (
        <>
          <h2 className="text-2xl font-bold mb-4">
            When you pray or worship, what does it feel like?
          </h2>
          <p className="text-sm text-gray-500 mb-6">Select 1-2 that resonate</p>

          <div className="space-y-3">
            {question1Options.map((option) => (
              <button
                key={option.id}
                onClick={() => handleMultiSelect(option.id)}
                className={`w-full p-4 text-left rounded-lg border-2 transition-all ${
                  answers.prayer_experience.includes(option.id)
                    ? 'border-blue-500 bg-blue-50'
                    : 'border-gray-200 hover:border-gray-300'
                }`}
              >
                {option.text}
              </button>
            ))}
          </div>
        </>
      )}

      {step === 2 && (
        <>
          <h2 className="text-2xl font-bold mb-4">
            Has there been a time when you felt close to God?
          </h2>

          <div className="space-y-3">
            {question2Options.map((option) => (
              <button
                key={option.id}
                onClick={() => handleSingleSelect(option.id)}
                className={`w-full p-4 text-left rounded-lg border-2 transition-all ${
                  answers.past_connection === option.id
                    ? 'border-blue-500 bg-blue-50'
                    : 'border-gray-200 hover:border-gray-300'
                }`}
              >
                {option.text}
              </button>
            ))}
          </div>
        </>
      )}

      <button
        onClick={handleNext}
        disabled={
          (step === 1 && answers.prayer_experience.length === 0) ||
          (step === 2 && !answers.past_connection)
        }
        className={`w-full mt-8 py-4 px-6 rounded-lg font-semibold text-white transition-all ${
          (step === 1 && answers.prayer_experience.length > 0) ||
          (step === 2 && answers.past_connection)
            ? 'bg-blue-600 hover:bg-blue-700'
            : 'bg-gray-300 cursor-not-allowed'
        }`}
      >
        Continue
      </button>
    </div>
  );
};

export default Phase2A_DistantFromGod;
```

## PHASE 2: Backend Persona Assignment Logic

### /server/utils/personaAssignment.js
```javascript
function assignPersona(userData) {
  const personas = [];
  
  // Primary struggle mapping
  if (userData.primary_struggle === 'distant_from_god' && 
      userData.depth_layer?.prayer_experience?.includes('empty_room')) {
    personas.push('seeker_in_void');
  } else if (userData.primary_struggle === 'distant_from_god') {
    personas.push('seeker_in_void');
  }
  
  if (userData.primary_struggle === 'wrestling_doubts' && 
      userData.depth_layer?.doubt_emotion === 'scared_losing_faith') {
    personas.push('doubter_in_crisis');
  } else if (userData.primary_struggle === 'wrestling_doubts') {
    personas.push('doubter_in_crisis');
  }
  
  if (userData.primary_struggle === 'feel_alone') {
    personas.push('isolated_wanderer');
  }
  
  if (userData.primary_struggle === 'guilt_shame') {
    personas.push('guilt_ridden_striver');
  }
  
  if (userData.primary_struggle === 'life_overwhelming') {
    personas.push('overwhelmed_survivor');
  }
  
  if (userData.primary_struggle === 'new_to_faith') {
    personas.push('hungry_beginner');
  }
  
  // Secondary persona modifiers (can stack)
  if (userData.behavioral_reality?.obstacles?.includes('lose_momentum')) {
    personas.push('momentum_breaker');
  }
  
  if (userData.behavioral_reality?.obstacles?.includes('compare_others') || 
      userData.behavioral_reality?.obstacles?.includes('doing_wrong')) {
    personas.push('comparison_captive');
  }
  
  // Return primary + secondary modifiers
  return {
    primary: personas[0] || 'seeker_in_void', // Default fallback
    modifiers: personas.slice(1)
  };
}

module.exports = { assignPersona };
```

```javascript
const anthropicService = require('../services/anthropicService');
const db = require('../services/databaseService');
const { buildAISystemPrompt } = require('../utils/aiPromptBuilder');

async function sendMessage(req, res) {
  try {
    const userId = req.user.id;
    const { message, conversationId } = req.body;

    // Get user persona
    const userPersona = await db.getUserPersona(userId);
    
    if (!userPersona) {
      return res.status(400).json({
        success: false,
        error: 'Please complete onboarding first'
      });
    }

    // Build system prompt based on persona
    const systemPrompt = buildAISystemPrompt(userPersona);

    // Get conversation history
    let conversation = null;
    if (conversationId) {
      conversation = await db.getConversation(conversationId);
    } else {
      // Create new conversation
      conversation = await db.createConversation(userId);
    }

    const conversationHistory = await db.getMessages(conversation.id);

    // Save user message
    await db.saveMessage(conversation.id, 'user', message);

    // Call Claude API with system prompt + history
    const aiResponse = await anthropicService.sendMessage({
      systemPrompt,
      conversationHistory,
      userMessage: message
    });

    // Save assistant message
    await db.saveMessage(conversation.id, 'assistant', aiResponse);

    res.status(200).json({
      success: true,
      conversationId: conversation.id,
      response: aiResponse
    });

  } catch (error) {
    console.error('Error in sendMessage:', error);
    res.status(500).json({
      success: false,
      error: 'Failed to send message'
    });
  }
}

module.exports = { sendMessage };
```

### /server/services/anthropicService.js
```javascript
const Anthropic = require('@anthropic-ai/sdk');

const anthropic = new Anthropic({
  apiKey: process.env.ANTHROPIC_API_KEY,
});

async function sendMessage({ systemPrompt, conversationHistory, userMessage }) {
  try {
    // Format conversation history for Claude API
    const messages = [
      ...conversationHistory.map(msg => ({
        role: msg.role,
        content: msg.content
      })),
      {
        role: 'user',
        content: userMessage
      }
    ];

    const response = await anthropic.messages.create({
      model: 'claude-sonnet-4-20250514',
      max_tokens: 1024,
      system: systemPrompt,
      messages: messages
    });

    return response.content[0].text;

  } catch (error) {
    console.error('Anthropic API error:', error);
    throw new Error('Failed to get AI response');
  }
}

module.exports = { sendMessage };
```
