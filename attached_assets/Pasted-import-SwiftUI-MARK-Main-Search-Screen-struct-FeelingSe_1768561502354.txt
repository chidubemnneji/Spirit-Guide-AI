import SwiftUI

// MARK: - Main Search Screen
struct FeelingSearchView: View {
    @State private var searchText: String = ""
    @State private var results: [Verse] = []
    @State private var bookmarks: [Bookmark] = []
    @State private var selectedVerse: Bookmark? = nil
    @State private var showReflectionCTA: Bool = false
    
    var body: some View {
        ZStack {
            Color(.systemGroupedBackground).ignoresSafeArea()
            
            VStack {
                // Search bar
                HStack {
                    TextField("Search by feeling or keyword…", text: $searchText, onCommit: {
                        performSearch()
                    })
                    .padding(12)
                    .background(Color(.systemBackground).opacity(0.9))
                    .cornerRadius(12)
                    .shadow(color: .black.opacity(0.05), radius: 6, y: 3)
                    
                    Button(action: { performSearch() }) {
                        Image(systemName: "magnifyingglass")
                            .foregroundColor(.green)
                    }
                    .padding(.leading, 6)
                }
                .padding()
                
                // Results list
                ScrollView {
                    VStack(spacing: 16) {
                        ForEach(results.prefix(5), id: \.id) { verse in
                            VerseCard(
                                verse: verse,
                                isBookmarked: bookmarks.contains(where: { $0.verseNumber == verse.number && $0.book == verse.book }),
                                onBookmark: {
                                    let bookmark = Bookmark(verseNumber: verse.number, text: verse.text, book: verse.book)
                                    if let index = bookmarks.firstIndex(where: { $0.verseNumber == verse.number && $0.book == verse.book }) {
                                        bookmarks.remove(at: index)
                                    } else {
                                        bookmarks.append(bookmark)
                                    }
                                    selectedVerse = bookmark
                                    showReflectionCTA = true
                                }
                            )
                            .animation(.easeOut(duration: 0.3), value: bookmarks)
                        }
                    }
                    .padding()
                }
            }
            
            // Reflection CTA overlay
            if showReflectionCTA, let verse = selectedVerse {
                ReflectionCTA(verse: verse) {
                    // User tapped "Reflect"
                    print("Start reflection for: \(verse.text)")
                    showReflectionCTA = false
                }
                .transition(.move(edge: .bottom).combined(with: .opacity))
                .animation(.spring(), value: showReflectionCTA)
            }
        }
    }
    
    // MARK: - Perform Search (Curated 3–5 verses)
    func performSearch() {
        let query = searchText.lowercased()
        
        // Simple mapping for demonstration
        let feelingMap: [String: [Verse]] = [
            "anxious": [
                Verse(number: 10, text: "Fear not, for I am with you; be not dismayed, for I am your God.", book: "Isaiah 41"),
                Verse(number: 6, text: "Do not be anxious about anything, but in everything by prayer let your requests be made known to God.", book: "Philippians 4"),
                Verse(number: 4, text: "I sought the Lord, and he answered me and delivered me from all my fears.", book: "Psalm 34")
            ],
            "fear": [
                Verse(number: 10, text: "The Lord is my light and my salvation; whom shall I fear?", book: "Psalm 27"),
                Verse(number: 9, text: "Have I not commanded you? Be strong and courageous. Do not be afraid; do not be discouraged.", book: "Joshua 1")
            ]
        ]
        
        // Pick mapped verses if query matches, else empty
        results = feelingMap.first { query.contains($0.key) }?.value ?? []
    }
}

// MARK: - Verse Model
struct Verse: Identifiable {
    var id = UUID()
    let number: Int
    let text: String
    let book: String
}

// MARK: - Bookmark Model
struct Bookmark: Identifiable, Equatable {
    var id = UUID()
    let verseNumber: Int
    let text: String
    let book: String
}
